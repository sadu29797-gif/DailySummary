### 技术难点突破

在学习过程中遇到并解决了几个技术难点：

```C
#define _CRT_SECURE_NO_WARNINGS 1
#include <stdio.h>

int Factorial_Error(int n) {
    if (n >= 1) {
        return n * Factorial_Error(n - 1);
    }
    return 1;
}

int Factorial_Correct(int n) {
    if (n < 0) {
        printf("错误：n不能为负数\n");
        return -1;
    }
    if (n == 0 || n == 1) {
        return 1;
    }
    return n * Factorial_Correct(n - 1);
}

int Factorial_Iteration(int n) {
    if (n < 0) {
        printf("错误：n不能为负数\n");
        return -1;
    }
    int result = 1;
    for (int i = 2; i <= n; i++) {
        result *= i;
    }
    return result;
}

int main() {
    int test1 = Factorial_Correct(5);
    if (test1 != -1) {
        printf("5的阶乘（递归）：%d\n", test1);
    }

    int test2 = Factorial_Iteration(10);
    if (test2 != -1) {
        printf("10的阶乘（迭代）：%d\n", test2);
    }

    Factorial_Correct(-3);

    return 0;
}

```
